// src/modules/leads/services/leadsApi.js
import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react';

export const leadsApi = createApi({
  reducerPath: 'leadsApi',
  baseQuery: fetchBaseQuery({ baseUrl: '/api' }),
  tagTypes: ['Leads'],
  endpoints: (builder) => ({
    getLeads: builder.query({
      query: () => 'leads',
      providesTags: (result) =>
        result ? [...result.map(({ id }) => ({ type: 'Leads', id })), 'Leads'] : ['Leads'],
    }),
    getLead: builder.query({
      query: (id) => `leads/${id}`,
      providesTags: (result, error, id) => [{ type: 'Leads', id }],
    }),
    createLead: builder.mutation({
      query: (payload) => ({
        url: 'leads',
        method: 'POST',
        body: payload,
      }),
      invalidatesTags: ['Leads'],
    }),
    updateLead: builder.mutation({
      query: ({ id, ...patch }) => ({
        url: `leads/${id}`,
        method: 'PUT',
        body: patch,
      }),
      invalidatesTags: (result, error, { id }) => [{ type: 'Leads', id }],
    }),
    deleteLead: builder.mutation({
      query: (id) => ({ url: `leads/${id}`, method: 'DELETE' }),
      invalidatesTags: ['Leads'],
    }),
    getLeadSchema: builder.query({
      query: () => 'modules/leads/form-schema',
    }),
  }),
});

export const {
  useGetLeadsQuery,
  useGetLeadQuery,
  useCreateLeadMutation,
  useUpdateLeadMutation,
  useDeleteLeadMutation,
  useGetLeadSchemaQuery,
} = leadsApi;

// src/modules/leads/renderer/DynamicFormRenderer.jsx
import React from 'react';
import { useGetLeadSchemaQuery, useCreateLeadMutation } from '@/modules/leads/services/leadsApi';
import UniversalFieldRenderer from '@/modules/leads/renderer/UniversalFieldRenderer';
import { Button } from '@/components/ui/button';
import { useForm } from 'react-hook-form';

export default function DynamicFormRenderer() {
  const { data: schema, isLoading } = useGetLeadSchemaQuery();
  const [createLead] = useCreateLeadMutation();
  const { handleSubmit, register, setValue } = useForm();

  if (isLoading) return <p>Loading form...</p>;

  const onSubmit = async (values) => {
    await createLead(values);
    // optionally show success toast
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
      {schema.sections.map((section) => (
        <div key={section.title} className="border p-4 rounded-md">
          <h3 className="text-lg font-semibold mb-2">{section.title}</h3>
          <div className="grid gap-4" style={{ gridTemplateColumns: `repeat(12, minmax(0, 1fr))` }}>
            {section.fields.map((field) => (
              <UniversalFieldRenderer
                key={field.name}
                field={field}
                register={register}
                setValue={setValue}
              />
            ))}
          </div>
        </div>
      ))}
      <Button type="submit">Submit</Button>
    </form>
  );
}

// src/modules/leads/renderer/UniversalFieldRenderer.jsx
import React from 'react';
import { Input } from '@/components/ui/input';
import { Select } from '@/components/ui/select';
import { Textarea } from '@/components/ui/textarea';
import { Switch } from '@/components/ui/switch';
import { Calendar } from '@/components/ui/calendar';
import { Label } from '@/components/ui/label';

const FIELD_COMPONENTS = {
  text: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Input {...register(field.name)} placeholder={field.placeholder} />
    </div>
  ),
  email: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Input type="email" {...register(field.name)} placeholder={field.placeholder} />
    </div>
  ),
  phone: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Input type="tel" {...register(field.name)} placeholder={field.placeholder} />
    </div>
  ),
  number: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Input type="number" {...register(field.name)} placeholder={field.placeholder} />
    </div>
  ),
  textarea: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Textarea {...register(field.name)} placeholder={field.placeholder} />
    </div>
  ),
  select: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Select {...register(field.name)}>
        {field.options?.map((opt) => (
          <Select.Option key={opt.value} value={opt.value}>
            {opt.label}
          </Select.Option>
        ))}
      </Select>
    </div>
  ),
  checkbox: ({ field, register }) => (
    <div className="flex items-center space-x-2">
      <Switch {...register(field.name)} />
      <Label>{field.label}</Label>
    </div>
  ),
  date: ({ field, register }) => (
    <div>
      <Label>{field.label}</Label>
      <Calendar {...register(field.name)} />
    </div>
  ),
  // add more mappings as needed
};

export default function UniversalFieldRenderer({ field, register, setValue }) {
  const Component = FIELD_COMPONENTS[field.type];
  if (!Component) return null;
  return <Component field={field} register={register} setValue={setValue} />;
}
